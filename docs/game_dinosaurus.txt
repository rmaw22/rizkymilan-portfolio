"use client";

import { useEffect, useRef, useState, useCallback } from "react";

const W = 680;
const H = 200;
const GND = 155;

type GameState = "idle" | "running" | "dead";

interface Obstacle {
  x: number;
  y: number;
  w: number;
  h: number;
}

interface Cloud {
  x: number;
  y: number;
  w: number;
}

interface Dino {
  x: number;
  y: number;
  w: number;
  h: number;
  vy: number;
  jumping: boolean;
}

export default function DinoRunner() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const stateRef = useRef<GameState>("idle");
  const dinoRef = useRef<Dino>({ x: 60, y: GND, w: 36, h: 40, vy: 0, jumping: false });
  const obstaclesRef = useRef<Obstacle[]>([]);
  const cloudsRef = useRef<Cloud[]>([]);
  const frameRef = useRef(0);
  const scoreRef = useRef(0);
  const hiScoreRef = useRef(0);
  const baseSpeedRef = useRef(5);
  const rafRef = useRef<number>(0);

  const [score, setScore] = useState(0);
  const [hiScore, setHiScore] = useState(0);
  const [gameState, setGameState] = useState<GameState>("idle");
  const [baseSpeed, setBaseSpeed] = useState(5);

  const getColors = useCallback(() => {
    const isDark =
      window.matchMedia &&
      window.matchMedia("(prefers-color-scheme: dark)").matches;
    return {
      bg: isDark ? "#1a1a1a" : "#ffffff",
      ground: isDark ? "#555555" : "#888888",
      dino: isDark ? "#cccccc" : "#333333",
      cactus: isDark ? "#5aaa99" : "#3a7a55",
      cloud: isDark ? "#444444" : "#dddddd",
      text: isDark ? "#aaaaaa" : "#555555",
    };
  }, []);

  const drawDino = useCallback(
    (ctx: CanvasRenderingContext2D, x: number, y: number, w: number, h: number, frame: number, state: GameState) => {
      const C = getColors();
      const f = Math.floor(frame / 8) % 2;
      ctx.fillStyle = C.dino;
      ctx.fillRect(x + 8, y - h, w - 8, h - 10);
      ctx.fillRect(x + 20, y - h - 8, 14, 10);
      ctx.fillStyle = C.bg;
      ctx.fillRect(x + 26, y - h - 6, 4, 4);
      ctx.fillStyle = C.dino;
      if (state === "running" || state === "idle") {
        if (f === 0) {
          ctx.fillRect(x + 10, y - 10, 8, 10);
          ctx.fillRect(x + 20, y - 6, 8, 6);
        } else {
          ctx.fillRect(x + 12, y - 6, 8, 6);
          ctx.fillRect(x + 22, y - 10, 8, 10);
        }
      } else {
        ctx.fillRect(x + 10, y - 10, 8, 10);
        ctx.fillRect(x + 22, y - 10, 8, 10);
      }
      ctx.fillRect(x + w - 10, y - h + 10, 12, 8);
    },
    [getColors]
  );

  const drawCactus = useCallback(
    (ctx: CanvasRenderingContext2D, obs: Obstacle) => {
      const C = getColors();
      ctx.fillStyle = C.cactus;
      ctx.fillRect(obs.x + obs.w / 2 - 6, obs.y - obs.h, 12, obs.h);
      ctx.fillRect(obs.x + obs.w / 2 - 18, obs.y - obs.h * 0.55, 12, obs.h * 0.3);
      ctx.fillRect(obs.x + obs.w / 2 + 6, obs.y - obs.h * 0.65, 12, obs.h * 0.35);
      ctx.fillRect(obs.x + obs.w / 2 - 18, obs.y - obs.h * 0.55 - obs.h * 0.3, 6, obs.h * 0.3);
      ctx.fillRect(obs.x + obs.w / 2 + 12, obs.y - obs.h * 0.65 - obs.h * 0.35, 6, obs.h * 0.35);
    },
    [getColors]
  );

  const drawCloud = useCallback(
    (ctx: CanvasRenderingContext2D, cl: Cloud) => {
      const C = getColors();
      ctx.fillStyle = C.cloud;
      ctx.fillRect(cl.x, cl.y, cl.w, 12);
      ctx.fillRect(cl.x + 10, cl.y - 8, cl.w - 20, 10);
    },
    [getColors]
  );

  const drawGround = useCallback(
    (ctx: CanvasRenderingContext2D, frame: number) => {
      const C = getColors();
      ctx.fillStyle = C.ground;
      ctx.fillRect(0, GND + 2, W, 2);
      for (let i = 0; i < 12; i++) {
        const rx = (frame * 1.2 + i * 57) % W;
        ctx.fillRect(rx, GND + 6, 18, 2);
      }
    },
    [getColors]
  );

  const rectCollide = (a: Obstacle, b: Obstacle) => {
    const pad = 8;
    return !(
      a.x + a.w - pad < b.x + pad ||
      a.x + pad > b.x + b.w - pad ||
      a.y - a.h + pad < b.y - b.h + pad ||
      a.y - pad > b.y + pad
    );
  };

  const draw = useCallback(
    (ctx: CanvasRenderingContext2D) => {
      const C = getColors();
      const state = stateRef.current;
      const dino = dinoRef.current;
      const frame = frameRef.current;

      ctx.clearRect(0, 0, W, H);
      ctx.fillStyle = C.bg;
      ctx.fillRect(0, 0, W, H);

      cloudsRef.current.forEach((cl) => drawCloud(ctx, cl));
      drawGround(ctx, frame);
      obstaclesRef.current.forEach((obs) => drawCactus(ctx, obs));
      drawDino(ctx, dino.x, dino.y, dino.w, dino.h, frame, state);

      ctx.fillStyle = C.text;
      ctx.font = "14px monospace";
      ctx.textAlign = "center";
      if (state === "idle") {
        ctx.fillText("Tekan Spasi / Klik untuk mulai", W / 2, H / 2 + 8);
      }
      if (state === "dead") {
        ctx.fillText("GAME OVER — Spasi / Klik untuk ulang", W / 2, H / 2 + 8);
      }
    },
    [drawCloud, drawGround, drawCactus, drawDino, getColors]
  );

  const gameOver = useCallback(() => {
    stateRef.current = "dead";
    setGameState("dead");
  }, []);

  const update = useCallback(() => {
    const dino = dinoRef.current;
    const frame = ++frameRef.current;
    const speed = baseSpeedRef.current + scoreRef.current * 0.003;

    dino.vy += 1.6;
    dino.y += dino.vy;
    if (dino.y >= GND) {
      dino.y = GND;
      dino.vy = 0;
      dino.jumping = false;
    }

    const spawnInterval = Math.max(60, 120 - scoreRef.current / 4);
    if (frame % spawnInterval === 0) {
      const h = 40 + Math.random() * 30;
      obstaclesRef.current.push({ x: W + 20, y: GND, w: 40, h });
    }
    if (frame % 90 === 0) {
      cloudsRef.current.push({ x: W + 20, y: 30 + Math.random() * 50, w: 80 + Math.random() * 40 });
    }

    obstaclesRef.current = obstaclesRef.current.filter((o) => {
      o.x -= speed;
      return o.x + o.w >= 0;
    });
    cloudsRef.current = cloudsRef.current.filter((cl) => {
      cl.x -= speed * 0.4;
      return cl.x + cl.w >= 0;
    });

    scoreRef.current++;
    if (scoreRef.current > hiScoreRef.current) {
      hiScoreRef.current = scoreRef.current;
      setHiScore(hiScoreRef.current);
    }
    setScore(scoreRef.current);

    const dinoRect: Obstacle = { x: dino.x, y: dino.y, w: dino.w - 8, h: dino.h };
    for (const obs of obstaclesRef.current) {
      if (rectCollide(dinoRect, obs)) {
        gameOver();
        return;
      }
    }
  }, [gameOver]);

  const loop = useCallback(
    (ctx: CanvasRenderingContext2D) => {
      if (stateRef.current === "running") update();
      draw(ctx);
      rafRef.current = requestAnimationFrame(() => loop(ctx));
    },
    [update, draw]
  );

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;
    rafRef.current = requestAnimationFrame(() => loop(ctx));
    return () => cancelAnimationFrame(rafRef.current);
  }, [loop]);

  const jump = useCallback(() => {
    if (stateRef.current === "idle" || stateRef.current === "dead") {
      stateRef.current = "running";
      setGameState("running");
      scoreRef.current = 0;
      frameRef.current = 0;
      obstaclesRef.current = [];
      cloudsRef.current = [];
      dinoRef.current = { x: 60, y: GND, w: 36, h: 40, vy: 0, jumping: false };
      setScore(0);
      return;
    }
    if (!dinoRef.current.jumping) {
      dinoRef.current.vy = -18;
      dinoRef.current.jumping = true;
    }
  }, []);

  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if (e.code === "Space" || e.code === "ArrowUp") {
        e.preventDefault();
        jump();
      }
    };
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  }, [jump]);

  const handleSpeedChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const v = Number(e.target.value);
    setBaseSpeed(v);
    baseSpeedRef.current = v;
  };

  return (
    <div style={{ fontFamily: "monospace", padding: "1.5rem 0" }}>
      <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 12 }}>
        <span style={{ fontSize: 13 }}>Score: <strong>{score}</strong></span>
        <span style={{ fontSize: 13 }}>Best: <strong>{hiScore}</strong></span>
      </div>

      <canvas
        ref={canvasRef}
        width={W}
        height={H}
        onClick={jump}
        style={{
          display: "block",
          width: "100%",
          height: "auto",
          borderRadius: 8,
          border: "1px solid #e5e7eb",
          cursor: "pointer",
        }}
      />

      <p style={{ textAlign: "center", fontSize: 13, color: "#888", marginTop: 12 }}>
        {gameState === "idle" && "Klik / Tekan Spasi atau ↑ untuk mulai"}
        {gameState === "running" && "Spasi / Klik / ↑ = Lompat"}
        {gameState === "dead" && "Klik / Tekan Spasi untuk main lagi"}
      </p>

      <div style={{ display: "flex", alignItems: "center", gap: 10, marginTop: 14 }}>
        <label style={{ fontSize: 13, color: "#888", whiteSpace: "nowrap" }}>
          Kecepatan awal
        </label>
        <input
          type="range"
          min={3}
          max={10}
          step={1}
          value={baseSpeed}
          onChange={handleSpeedChange}
          style={{ flex: 1 }}
        />
        <span style={{ fontSize: 13, fontWeight: 500, minWidth: 20 }}>{baseSpeed}</span>
      </div>
    </div>
  );
}
